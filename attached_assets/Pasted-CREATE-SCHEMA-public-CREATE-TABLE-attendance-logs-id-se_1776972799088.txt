CREATE SCHEMA "public";

CREATE TABLE "attendance_logs" (

"id" serial PRIMARY KEY,

"company_id" integer NOT NULL,

"employee_id" integer NOT NULL,

"type" text NOT NULL,

"timestamp" timestamp with time zone DEFAULT now() NOT NULL,

"notes" text

);

CREATE TABLE "audit_logs" (

"id" serial PRIMARY KEY,

"company_id" integer,

"user_id" integer,

"user_email" text,

"action" text NOT NULL,

"resource" text,

"resource_id" text,

"details" text,

"ip_address" text,

"timestamp" timestamp with time zone DEFAULT now() NOT NULL,

"user_agent" text,

"device" text

);

CREATE TABLE "companies" (

"id" serial PRIMARY KEY,

"name" text NOT NULL,

"email" text NOT NULL CONSTRAINT "companies_email_unique" UNIQUE,

"created_at" timestamp with time zone DEFAULT now() NOT NULL

);

CREATE TABLE "employees" (

"id" serial PRIMARY KEY,

"company_id" integer NOT NULL,

"document_number" text NOT NULL CONSTRAINT "employees_document_number_unique" UNIQUE,

"name" text NOT NULL,

"position" text NOT NULL,

"department" text NOT NULL,

"status" text DEFAULT 'active' NOT NULL,

"email" text,

"phone" text,

"pin_hash" text,

"created_at" timestamp with time zone DEFAULT now() NOT NULL,

"updated_at" timestamp with time zone DEFAULT now() NOT NULL

);

CREATE TABLE "users" (

"id" serial PRIMARY KEY,

"company_id" integer,

"email" text NOT NULL CONSTRAINT "users_email_unique" UNIQUE,

"password_hash" text NOT NULL,

"name" text NOT NULL,

"role" text DEFAULT 'employee' NOT NULL,

"created_at" timestamp with time zone DEFAULT now() NOT NULL,

"updated_at" timestamp with time zone DEFAULT now() NOT NULL

);

CREATE TABLE "work_schedule" (

"id" serial PRIMARY KEY,

"company_id" integer,

"start_time" text DEFAULT '08:00' NOT NULL,

"end_time" text DEFAULT '17:00' NOT NULL,

"work_days" text[] DEFAULT '{mon,tue,wed,thu,fri}' NOT NULL,

"late_tolerance_minutes" integer DEFAULT 15 NOT NULL

);

ALTER TABLE "attendance_logs" ADD CONSTRAINT "attendance_logs_company_id_companies_id_fk" FOREIGN KEY ("company_id") REFERENCES "companies"("id") ON DELETE CASCADE;

ALTER TABLE "attendance_logs" ADD CONSTRAINT "attendance_logs_employee_id_employees_id_fk" FOREIGN KEY ("employee_id") REFERENCES "employees"("id") ON DELETE CASCADE;

ALTER TABLE "audit_logs" ADD CONSTRAINT "audit_logs_company_id_companies_id_fk" FOREIGN KEY ("company_id") REFERENCES "companies"("id") ON DELETE SET NULL;

ALTER TABLE "audit_logs" ADD CONSTRAINT "audit_logs_user_id_users_id_fk" FOREIGN KEY ("user_id") REFERENCES "users"("id") ON DELETE SET NULL;

ALTER TABLE "employees" ADD CONSTRAINT "employees_company_id_companies_id_fk" FOREIGN KEY ("company_id") REFERENCES "companies"("id") ON DELETE CASCADE;

ALTER TABLE "users" ADD CONSTRAINT "users_company_id_companies_id_fk" FOREIGN KEY ("company_id") REFERENCES "companies"("id") ON DELETE CASCADE;

ALTER TABLE "work_schedule" ADD CONSTRAINT "work_schedule_company_id_companies_id_fk" FOREIGN KEY ("company_id") REFERENCES "companies"("id") ON DELETE CASCADE;

CREATE UNIQUE INDEX "attendance_logs_pkey" ON "attendance_logs" ("id");

CREATE UNIQUE INDEX "audit_logs_pkey" ON "audit_logs" ("id");

CREATE UNIQUE INDEX "companies_email_unique" ON "companies" ("email");

CREATE UNIQUE INDEX "companies_pkey" ON "companies" ("id");

CREATE UNIQUE INDEX "employees_document_number_unique" ON "employees" ("document_number");

CREATE UNIQUE INDEX "employees_pkey" ON "employees" ("id");

CREATE UNIQUE INDEX "users_email_unique" ON "users" ("email");

CREATE UNIQUE INDEX "users_pkey" ON "users" ("id");

CREATE UNIQUE INDEX "work_schedule_pkey" ON "work_schedule" ("id");
